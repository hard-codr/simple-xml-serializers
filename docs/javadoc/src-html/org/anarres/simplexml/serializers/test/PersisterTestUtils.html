005package org.anarres.simplexml.serializers.test;
006
007import java.io.IOException;
008import javax.annotation.Nonnull;
009import org.anarres.simplexml.factory.PersisterFactory;
010import org.anarres.simplexml.factory.PersisterUtils;
011import org.simpleframework.xml.core.Persister;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014import static org.junit.Assert.*;
015
016/**
017 *
018 * @author shevek
019 */
020public class PersisterTestUtils {
021
022    private static final Logger LOG = LoggerFactory.getLogger(PersisterTestUtils.class);
023
024    @SuppressWarnings({"rawtypes", "unchecked"})
025    @Nonnull
026    public static <T> T testSerialization(@Nonnull T in) throws IOException {
027        return (T) testSerialization((Class) in.getClass(), in);
028    }
029
030    @SuppressWarnings({"rawtypes", "unchecked"})
031    @Nonnull
032    public static <T> T testSerialization(@Nonnull Persister persister, @Nonnull T in) throws IOException {
033        return (T) testSerialization(persister, (Class) in.getClass(), in);
034    }
035
036    @Nonnull
037    public static <T> T testSerialization(@Nonnull Class<T> type, @Nonnull T in) throws IOException {
038        Persister persister = PersisterFactory.newDefaultPersister();
039        return testSerialization(persister, type, in);
040    }
041
042    @Nonnull
043    public static <T> T testSerialization(@Nonnull Persister persister, @Nonnull Class<T> type, @Nonnull T in) throws IOException {
044
045        LOG.info("Testing serialization of " + type);
046        LOG.info("In:  " + in);
047        assertNotNull(in);
048        byte[] data = PersisterUtils.save(persister, in, false);
049        LOG.info("Xml:  " + new String(data));
050        T out = PersisterUtils.load(persister, data, type, false);
051        LOG.info("Out: " + out);
052        assertNotNull(out);
053        return out;
054    }
055
056    public static <T> void testEquality(T a, T b, boolean result) {
057        LOG.info("Testing equality of " + a + (result ? " == " : " != ") + b);
058        if (result)
059            assertEquals(a + " == " + b, a, b);
060        else
061            assertNotEquals(a + " != " + b, a, b);
062        if (result)
063            assertEquals(a.hashCode(), b.hashCode());
064    }
065}

001package org.anarres.simplexml.serializers.jdk;
002
003import com.google.auto.service.AutoService;
004import com.google.common.collect.Maps;
005import java.nio.charset.Charset;
006import java.util.Collections;
007import java.util.Map;
008import org.anarres.simplexml.serializers.common.TransformFactory;
009import org.simpleframework.xml.transform.Transform;
010
011/**
012 *
013 * @author shevek
014 */
015@AutoService(TransformFactory.class)
016public class CharsetTransform implements Transform<Charset>, TransformFactory {
017
018    @Override
019    public Iterable<? extends Map.Entry<? extends Class<?>, ? extends Transform<?>>> newTransforms() {
020        return Collections.singleton(Maps.immutableEntry(Charset.class, this));
021    }
022
023    @Override
024    public Charset read(String string) throws Exception {
025        return Charset.forName(string);
026    }
027
028    @Override
029    public String write(Charset t) throws Exception {
030        return t.name();
031    }
032}

005package org.anarres.simplexml.serializers.spring.test;
006
007import java.io.ByteArrayInputStream;
008import java.io.ByteArrayOutputStream;
009import javax.annotation.Nonnull;
010import org.anarres.simplexml.serializers.spring.remote.SimpleXmlInvocationResult;
011import org.anarres.simplexml.serializers.spring.remote.SimpleXmlInvokerClientConfiguration;
012import org.anarres.simplexml.serializers.spring.remote.SimpleXmlInvokerRequestExecutor;
013import org.springframework.test.web.servlet.MockMvc;
014import org.springframework.test.web.servlet.ResultActions;
015import org.springframework.test.web.servlet.request.MockHttpServletRequestBuilder;
016import static org.springframework.test.web.servlet.request.MockMvcRequestBuilders.*;
017import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.*;
018
019/**
020 *
021 * @author shevek
022 */
023public class MockMvcSimpleXmlInvokerRequestExecutor extends SimpleXmlInvokerRequestExecutor {
024
025    private final MockMvc mvc;
026
027    public MockMvcSimpleXmlInvokerRequestExecutor(@Nonnull MockMvc mvc) {
028        this.mvc = mvc;
029    }
030
031    @Override
032    protected SimpleXmlInvocationResult doExecuteRequest(SimpleXmlInvokerClientConfiguration config, ByteArrayOutputStream baos) throws Exception {
033        MockHttpServletRequestBuilder builder = post(config.getServiceUrl());
034        builder.content(baos.toByteArray());
035        ResultActions result = mvc.perform(builder);
036        result.andExpect(status().isOk());
037        byte[] responseBody = result.andReturn().getResponse().getContentAsByteArray();
038        return readSimpleXmlInvocationResult(config, new ByteArrayInputStream(responseBody));
039    }
040}

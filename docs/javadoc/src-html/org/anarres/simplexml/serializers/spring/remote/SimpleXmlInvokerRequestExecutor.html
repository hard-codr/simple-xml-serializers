108    /**
109     * Execute the given request through a standard J2SE HttpURLConnection.
110     * <p>
111     * This method implements the basic processing workflow:
112     * The actual work happens in this class's template methods.
113     *
114     * @see #openConnection
115     * @see #prepareConnection
116     * @see #writeRequestBody
117     * @see #validateResponse
118     * @see #readResponseBody
119     */
120    public final SimpleXmlInvocationResult executeRequest(SimpleXmlInvokerClientConfiguration config, SimpleXmlInvocation invocation) throws Exception {
121        ByteArrayOutputStream baos = toByteArrayOutputStream(config, invocation);
122        if (LOG.isDebugEnabled()) {
123            LOG.debug("Sending HTTP invoker request for service at [" + config.getServiceUrl()
124                    + "], with size " + baos.size());
125        }
126        return doExecuteRequest(config, baos);
127    }

129    /**
130     * Serialize the given SimpleXmlInvocation into a ByteArrayOutputStream.
131     *
132     * @param invocation the SimpleXmlInvocation object
133     * @return a ByteArrayOutputStream with the serialized SimpleXmlInvocation
134     * @throws IOException if thrown by I/O methods
135     */
136    protected ByteArrayOutputStream toByteArrayOutputStream(SimpleXmlInvokerClientConfiguration config, SimpleXmlInvocation invocation) throws Exception {
137        ByteArrayOutputStream baos = new ByteArrayOutputStream(SERIALIZED_INVOCATION_BYTE_ARRAY_INITIAL_SIZE);
138        writeSimpleXmlInvocation(config, invocation, baos);
139        baos.write('\n');   // Makes debugging come out nicer with CommonsLogWriter.
140        return baos;
141    }
142
143    private void writeSimpleXmlInvocation(SimpleXmlInvokerClientConfiguration config, SimpleXmlInvocation invocation, OutputStream baos) throws Exception {
144        /*
145         if (LOG.isDebugEnabled()) {
146         Writer writer = new Slf4jLogWriter(LOG);
147         WriterOutputStream stream = new WriterOutputStream(writer);
148         baos = new TeeOutputStream(baos, stream);
149         }
150         */
151        config.getSerializer().write(invocation, baos);
152        baos.flush();
153    }

155    @Nonnull
156    protected SimpleXmlInvocationResult doExecuteRequest(
157            SimpleXmlInvokerClientConfiguration config, ByteArrayOutputStream baos)
158            throws Exception {
159
160        HttpURLConnection con = openConnection(config);
161        try {
162            prepareConnection(con, baos.size());
163            writeRequestBody(config, con, baos);
164            validateResponse(config, con);
165            InputStream responseBody = readResponseBody(config, con);
166
167            return readSimpleXmlInvocationResult(config, responseBody);
168        } finally {
169            con.disconnect();
170        }
171    }
172
173    @Nonnull
174    protected SimpleXmlInvocationResult readSimpleXmlInvocationResult(@Nonnull SimpleXmlInvokerClientConfiguration config, @Nonnull InputStream responseBody) throws Exception {
175        byte[] data = ByteStreams.toByteArray(responseBody);
176        LOG.info("Response is " + new String(data, Charsets.ISO_8859_1));   // Random 8-bit charset.
177        responseBody = new ByteArrayInputStream(data);
178        /*
179         if (false && LOG.isDebugEnabled()) {
180         Writer writer = new Slf4jLogWriter(LOG);
181         WriterOutputStream stream = new WriterOutputStream(writer);
182         responseBody = new TeeInputStream(responseBody, stream, true);
183         }
184         */
185        // LOG.debug("About to read response.");
186        try {
187            SimpleXmlInvocationResult result = config.getSerializer().read(SimpleXmlInvocationResult.class, responseBody, true);
188            LOG.info("Result is " + result);
189            return result;
190        } finally {
191            Closeables.close(responseBody, true);
192            // LOG.debug("Done reading response.");
193        }
194    }

196    /**
197     * Open an HttpURLConnection for the given remote invocation request.
198     *
199     * @param config the HTTP invoker configuration that specifies the
200     * target service
201     * @return the HttpURLConnection for the given request
202     * @throws IOException if thrown by I/O methods
203     * @see java.net.URL#openConnection()
204     */
205    protected HttpURLConnection openConnection(SimpleXmlInvokerClientConfiguration config) throws IOException {
206        URLConnection con = new URL(config.getServiceUrl()).openConnection();
207        if (!(con instanceof HttpURLConnection))
208            throw new IOException("Service URL [" + config.getServiceUrl() + "] is not an HTTP URL");
209        return (HttpURLConnection) con;
210    }

212    /**
213     * Prepare the given HTTP connection.
214     * <p>
215     * The default implementation specifies POST as method,
216     * "application/x-java-serialized-object" as "Content-Type" header,
217     * and the given content length as "Content-Length" header.
218     *
219     * @param connection the HTTP connection to prepare
220     * @param contentLength the length of the content to send
221     * @throws IOException if thrown by HttpURLConnection methods
222     * @see java.net.HttpURLConnection#setRequestMethod
223     * @see java.net.HttpURLConnection#setRequestProperty
224     */
225    protected void prepareConnection(HttpURLConnection connection, int contentLength) throws IOException {
226        if (this.connectTimeout >= 0)
227            connection.setConnectTimeout(this.connectTimeout);
228        if (this.readTimeout >= 0)
229            connection.setReadTimeout(this.readTimeout);
230        connection.setDoOutput(true);
231        connection.setRequestMethod(HTTP_METHOD_POST);
232        connection.setRequestProperty(HTTP_HEADER_CONTENT_TYPE, getContentType());
233        connection.setRequestProperty(HTTP_HEADER_CONTENT_LENGTH, Integer.toString(contentLength));
234        LocaleContext locale = LocaleContextHolder.getLocaleContext();
235        if (locale != null)
236            connection.setRequestProperty(HTTP_HEADER_ACCEPT_LANGUAGE, StringUtils.toLanguageTag(locale.getLocale()));
237        if (isAcceptGzipEncoding())
238            connection.setRequestProperty(HTTP_HEADER_ACCEPT_ENCODING, ENCODING_GZIP);
239    }

241    /**
242     * Set the given serialized remote invocation as request body.
243     * <p>
244     * The default implementation simply write the serialized invocation to the
245     * HttpURLConnection's OutputStream. This can be overridden, for example, to write
246     * a specific encoding and potentially set appropriate HTTP request headers.
247     *
248     * @param config the HTTP invoker configuration that specifies the target service
249     * @param con the HttpURLConnection to write the request body to
250     * @param baos the ByteArrayOutputStream that contains the serialized
251     * SimpleXmlInvocation object
252     * @throws IOException if thrown by I/O methods
253     * @see java.net.HttpURLConnection#getOutputStream()
254     * @see java.net.HttpURLConnection#setRequestProperty
255     */
256    protected void writeRequestBody(
257            SimpleXmlInvokerClientConfiguration config, HttpURLConnection con, ByteArrayOutputStream baos)
258            throws IOException {
259        baos.writeTo(con.getOutputStream());
260    }

262    /**
263     * Validate the given response as contained in the HttpURLConnection object,
264     * throwing an exception if it does not correspond to a successful HTTP response.
265     * <p>
266     * Default implementation rejects any HTTP status code beyond 2xx, to avoid
267     * parsing the response body and trying to deserialize from a corrupted stream.
268     *
269     * @param config the HTTP invoker configuration that specifies the target service
270     * @param con the HttpURLConnection to validate
271     * @throws IOException if validation failed
272     * @see java.net.HttpURLConnection#getResponseCode()
273     */
274    protected void validateResponse(SimpleXmlInvokerClientConfiguration config, HttpURLConnection con)
275            throws IOException {
276
277        int responseCode = con.getResponseCode();
278        if (responseCode >= 300) {
279            throw new IOException(
280                    "Did not receive successful HTTP response: status code = " + responseCode
281                    + ", status message = [" + con.getResponseMessage() + "]");
282        }
283    }
284
285    /**
286     * Extract the response body from the given executed remote invocation
287     * request.
288     * <p>
289     * The default implementation simply reads the serialized invocation
290     * from the HttpURLConnection's InputStream. If the response is recognized
291     * as GZIP response, the InputStream will get wrapped in a GZIPInputStream.
292     *
293     * @param config the HTTP invoker configuration that specifies the target service
294     * @param con the HttpURLConnection to read the response body from
295     * @return an InputStream for the response body
296     * @throws IOException if thrown by I/O methods
297     * @see #isGzipResponse
298     * @see java.util.zip.GZIPInputStream
299     * @see java.net.HttpURLConnection#getInputStream()
300     * @see java.net.HttpURLConnection#getHeaderField(int)
301     * @see java.net.HttpURLConnection#getHeaderFieldKey(int)
302     */
303    protected InputStream readResponseBody(SimpleXmlInvokerClientConfiguration config, HttpURLConnection con)
304            throws IOException {
305
306        if (isGzipResponse(con)) {
307            // GZIP response found - need to unzip.
308            return new GZIPInputStream(con.getInputStream());
309        } else {
310            // Plain response found.
311            return con.getInputStream();
312        }
313    }

315    /**
316     * Determine whether the given response is a GZIP response.
317     * <p>
318     * Default implementation checks whether the HTTP "Content-Encoding"
319     * header contains "gzip" (in any casing).
320     *
321     * @param con the HttpURLConnection to check
322     */
323    protected boolean isGzipResponse(HttpURLConnection con) {
324        String encodingHeader = con.getHeaderField(HTTP_HEADER_CONTENT_ENCODING);
325        return (encodingHeader != null && encodingHeader.toLowerCase().contains(ENCODING_GZIP));
326    }

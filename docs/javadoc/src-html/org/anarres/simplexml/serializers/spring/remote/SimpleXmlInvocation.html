005package org.anarres.simplexml.serializers.spring.remote;
006
007import com.google.common.annotations.VisibleForTesting;
008import java.io.Serializable;
009import java.util.Arrays;
010import java.util.Iterator;
011import java.util.Map;
012import javax.annotation.CheckForNull;
013import javax.annotation.Nonnull;
014import org.anarres.simplexml.factory.PersisterUtils;
015import org.aopalliance.intercept.MethodInvocation;
016import org.simpleframework.xml.Attribute;
017import org.simpleframework.xml.ElementArray;
018import org.simpleframework.xml.ElementMap;
019import org.simpleframework.xml.Root;
020import org.springframework.remoting.support.RemoteInvocation;
021
022/**
023 *
024 * @author shevek
025 */
026@Root(name = "call")
027public class SimpleXmlInvocation extends RemoteInvocation {
028
029    @VisibleForTesting
030    /* pp */ static final Object[] EMPTY_OBJECT_ARRAY = new Object[0];
031    @VisibleForTesting
032    /* pp */ static final Class<?>[] EMPTY_CLASS_ARRAY = new Class<?>[0];
033
034    @Nonnull
035    private static Class<?>[] unwrap(@CheckForNull Class<?>[] params) {
036        if (params == null)
037            return EMPTY_CLASS_ARRAY;
038        return params;
039    }
040
041    @Nonnull
042    private static Object[] unwrap(@CheckForNull Object[] args) {
043        if (args == null)
044            return EMPTY_OBJECT_ARRAY;
045        for (int i = 0; i < args.length; i++)
046            args[i] = PersisterUtils.unwrap(args[i]);
047        return args;
048    }
049
050    public SimpleXmlInvocation(
051            @Nonnull @Attribute(name = "method") String methodName,
052            @Nonnull @ElementArray(name = "params", entry = "param") Class<?>[] parameterTypes,
053            @Nonnull @ElementArray(name = "args", entry = "arg") Object[] arguments) {
054        super(methodName, unwrap(parameterTypes), unwrap(arguments));
055    }
056
057    public SimpleXmlInvocation(MethodInvocation methodInvocation) {
058        super(methodInvocation);
059    }
060
061    @Override
062    @Nonnull
063    @Attribute(name = "method")
064    public String getMethodName() {
065        return super.getMethodName();
066    }
067
068    @Override
069    @Nonnull
070    @ElementArray(name = "params", entry = "param", required = false, empty = false)
071    public Class<?>[] getParameterTypes() {
072        return super.getParameterTypes();
073    }
074
075    @Nonnull
076    @ElementArray(name = "args", entry = "arg", required = false, empty = false)
077    public Object[] getArgumentsWrapped() {
078        Object[] args = super.getArguments();
079        args = Arrays.copyOf(args, args.length);
080        for (int i = 0; i < args.length; i++)
081            args[i] = PersisterUtils.wrap(args[i]);
082        return args;
083    }
084
085    /** For serialization only. */
086    @ElementMap(name = "attrs", key = "key", value = "value", attribute = true, required = false)
087    public Map<String, ? extends Object> getAttributeMap() {
088        return super.getAttributes();
089    }
090
091    /** For serialization only. */
092    @ElementMap(name = "attrs", key = "key", value = "value", attribute = true, required = false)
093    public void setAttributeMap(Map<String, ? extends Object> attributes) {
094        Iterator<? extends Map.Entry<String, ?>> it = attributes.entrySet().iterator();
095        while (it.hasNext()) {
096            Map.Entry<String, ? extends Object> e = it.next();
097            if (e.getValue() != null)
098                if (!(e.getValue() instanceof Serializable))
099                    throw new IllegalArgumentException("Not serializable: " + e.getValue().getClass());
100        }
101        super.setAttributes((Map<String, Serializable>) attributes);
102    }
103}

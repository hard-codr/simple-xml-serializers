005package org.anarres.simplexml.serializers.spring.remote;
006
007import java.io.ByteArrayOutputStream;
008import java.io.IOException;
009import javax.servlet.ServletException;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import org.springframework.web.HttpRequestHandler;
013import org.springframework.web.HttpRequestMethodNotSupportedException;
014import org.springframework.web.util.NestedServletException;
015
016/**
017 * The primary bean for the RPC server.
018 *
019 * @author shevek
020 */
021public class SimpleXmlServiceExporter extends SimpleXmlExporter implements HttpRequestHandler {
022
023    @Override
024    public void handleRequest(HttpServletRequest request, HttpServletResponse response)
025            throws ServletException, IOException {
026
027        if (!"POST".equals(request.getMethod())) {
028            throw new HttpRequestMethodNotSupportedException(request.getMethod(),
029                    new String[]{"POST"}, "SimpleXmlServiceExporter only supports POST requests");
030        }
031
032        // request.getHeader(SimpleXmlInvokerRequestExecutor.HTTP_HEADER_ACCEPT_ENCODING, SimpleXmlInvokerRequestExecutor.ENCODING_GZIP);
033        try {
034            ByteArrayOutputStream baos = new ByteArrayOutputStream();
035            invoke(request.getInputStream(), baos);
036            response.setContentType(SimpleXmlInvokerRequestExecutor.CONTENT_TYPE_SIMPLEXML);
037            response.setContentLength(baos.size());
038            baos.writeTo(response.getOutputStream());
039        } catch (Throwable e) {
040            throw new NestedServletException("SimpleXML skeleton invocation failed", e);
041        }
042    }
043}

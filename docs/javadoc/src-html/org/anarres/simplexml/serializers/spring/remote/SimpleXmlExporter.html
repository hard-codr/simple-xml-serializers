005package org.anarres.simplexml.serializers.spring.remote;
006
007import com.google.common.io.Closeables;
008import java.io.InputStream;
009import java.io.OutputStream;
010import javax.annotation.CheckForNull;
011import javax.annotation.Nonnull;
012import org.anarres.simplexml.factory.PersisterFactory;
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015import org.simpleframework.xml.Serializer;
016import org.springframework.beans.factory.InitializingBean;
017import org.springframework.beans.factory.annotation.Autowired;
018import org.springframework.remoting.support.RemoteExporter;
019import org.springframework.util.Assert;
020
021/**
022 *
023 * @author shevek
024 */
025public class SimpleXmlExporter extends RemoteExporter implements InitializingBean {
026
027    private static final Logger LOG = LoggerFactory.getLogger(SimpleXmlExporter.class);
028    @Nonnull
029    private Serializer serializer = PersisterFactory.newDefaultPersister();
030    @CheckForNull
031    private Logger debugLogger = LOG;
032    private boolean close = false;
033    private Object target;
034
035    @Autowired(required = false)
036    public void setSerializer(@Nonnull Serializer serializer) {
037        this.serializer = serializer;
038    }
039
040    public void setDebug(boolean debug) {
041        this.debugLogger = (debug ? LOG : null);
042    }
043
044    public void setClose(boolean close) {
045        this.close = close;
046    }
047
048    @Override
049    public void afterPropertiesSet() {
050        prepare();
051    }
052
053    /**
054     * Initialize this exporter.
055     */
056    public void prepare() {
057        target = getProxyForService();
058        Assert.notNull(target, "Failed to create proxy target for " + getServiceInterface());
059    }
060
061    /**
062     * Perform an invocation on the exported object.
063     *
064     * @param inputStream the request stream
065     * @param outputStream the response stream
066     * @throws Throwable if invocation failed
067     */
068    public void invoke(@Nonnull InputStream inputStream, @Nonnull OutputStream outputStream) throws Throwable {
069        Assert.notNull(target, "No target object available.");
070        Assert.notNull(target.getClass(), "Target object has no class.");
071
072        ClassLoader originalClassLoader = overrideThreadContextClassLoader();
073
074        try {
075            /*
076             Flushable debugFlushable = null;
077             if (this.debugLogger != null && this.debugLogger.isDebugEnabled()) {
078             Writer debugWriter = new Slf4jLogWriter(this.debugLogger);
079             OutputStream debugOutput = new WriterOutputStream(debugWriter);
080             inputStream = new TeeInputStream(inputStream, debugOutput, true);
081             outputStream = new TeeOutputStream(outputStream, debugOutput);
082             debugFlushable = debugOutput;
083             }
084             */
085
086            SimpleXmlInvocation invocation = serializer.read(SimpleXmlInvocation.class, inputStream, false);
087            /*
088             if (debugFlushable != null)
089             debugFlushable.flush();
090             */
091            SimpleXmlInvocationResult result = new SimpleXmlInvocationResult();
092            try {
093                Object value = invocation.invoke(target);
094                if (LOG.isDebugEnabled())
095                    LOG.debug("Remote invocation returned value: " + value);
096                result.setValue(value);
097            } catch (Throwable t) {
098                if (LOG.isDebugEnabled())
099                    LOG.debug("Remote invocation caught error: " + t, t);
100                result.setThrowable(t);
101            }
102            if (LOG.isDebugEnabled())
103                LOG.debug("Remote invocation result is " + result);
104            serializer.write(result, outputStream);
105            outputStream.write('\n');   // Makes debugging come out more nicely with CommonsLogWriter.
106            outputStream.flush();
107        } finally {
108            resetThreadContextClassLoader(originalClassLoader);
109            if (close) {
110                Closeables.close(inputStream, true);
111                Closeables.close(outputStream, true);
112            }
113        }
114    }
115}

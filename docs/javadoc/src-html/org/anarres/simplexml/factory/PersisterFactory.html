035/**
036 * The main factory class for a {@link Persister}.
037 *
038 * To use this project:
039 * <ol>
040 * <li>Create a PersisterFactory
041 * <li>Add {@link ConverterFactory ConverterFactories} and {@link TransformFactory TransformFactories} to it.
042 * <li>Call newPersister()
043 * </ol>
044 *
045 * Various shortcuts for this process are provided in the case that you are using
046 * ServiceLoader, Spring Framework or other component discovery mechanisms.
047 *
048 * @author shevek
049 */
050public class PersisterFactory {
051
052    private static final Logger LOG = LoggerFactory.getLogger(PersisterFactory.class);
053    public static final String ID = "_xid";
054    public static final String REF = "_xref";
055
056    private final List<ConverterFactory> converterFactories = new ArrayList<ConverterFactory>();
057
058    public void addConverters(@Nonnull Iterable<? extends ConverterFactory> converterFactories) {
059        Iterables.addAll(this.converterFactories, converterFactories);
060    }
061
062    public void addConverters(@Nonnull ConverterFactory... converterFactories) {
063        addConverters(Arrays.asList(converterFactories));
064    }
065
066    private static class SuperClassLookupRegistry extends Registry {
067
068        private final List<Class<?>> superclasses = new ArrayList<Class<?>>();
069
070        @Override
071        @SuppressWarnings("rawtypes")
072        public Converter lookup(Class type) throws Exception {
073            Converter converter = super.lookup(type);
074            if (converter == null) {
075                if (Type.class.isAssignableFrom(type))
076                    return TypeConverter.getInstance();
077                else if (Charset.class.isAssignableFrom(type))
078                    return super.lookup(Charset.class);
079                // This cannot be Serializable because that would cause ArrayList etc to get b0rked.
080                else if (Throwable.class.isAssignableFrom(type)) // Includes Throwable.
081                    return super.lookup(Serializable.class);
082                // else if (TBase.class.isAssignableFrom(type)) return super.lookup(TBase.class);
083                // bind(type, converter);
084            }
085            // LOG.info("Convert " + type + " using " + converter, new Exception());
086            return converter;
087        }
088    }
089
090    private static void addConverters(@Nonnull Registry registry, @Nonnull Serializer serializer, @Nonnull Iterable<? extends ConverterFactory> factories) {
091        // LOG.info("Loading ConverterFactory instances.");
092        for (ConverterFactory factory : factories) {
093            // LOG.info("Factory " + factory);
094            for (Map.Entry<? extends Class<?>, ? extends Converter<?>> e : factory.newConverters(serializer)) {
095                // LOG.info("Register " + e.getKey() + " -> " + e.getValue());
096                try {
097                    registry.bind(e.getKey(), e.getValue());
098                } catch (Exception ex) {
099                    LOG.error("Failed to bind " + e.getKey() + " to " + e.getValue(), ex);
100                }
101            }
102        }
103    }
104
105    private final List<TransformFactory> transformFactories = new ArrayList<TransformFactory>();
106
107    public void addTransforms(@Nonnull Iterable<? extends TransformFactory> transformFactories) {
108        Iterables.addAll(this.transformFactories, transformFactories);
109    }
110
111    public void addTransforms(@Nonnull TransformFactory... transformFactories) {
112        addTransforms(Arrays.asList(transformFactories));
113    }
114
115    private static class SuperClassLookupRegistryMatcher extends RegistryMatcher {
116
117        @Override
118        @SuppressWarnings("rawtypes")
119        public Transform match(Class type) throws Exception {
120            Transform transform = super.match(type);
121            if (transform == null) {
122                if (Type.class.isAssignableFrom(type))
123                    transform = TypeTransform.getInstance();
124                else if (Throwable.class.isAssignableFrom(type))
125                    transform = SerializableTransform.getInstance();
126                else if (Charset.class.isAssignableFrom(type))
127                    return super.match(Charset.class);
128                // bind(type, transform);
129            }
130            // LOG.info("Transform " + type + " using " + transform, new Exception());
131            return transform;
132        }
133    }
134
135    private static void addTransformers(@Nonnull RegistryMatcher matcher, @Nonnull Serializer serializer, @Nonnull Iterable<? extends TransformFactory> factories) {
136        for (TransformFactory factory : factories) {
137            // LOG.info("Factory " + factory);
138            for (Map.Entry<? extends Class<?>, ? extends Transform<?>> e : factory.newTransforms()) {
139                // LOG.info("Register " + e.getKey() + " -> " + e.getValue());
140                try {
141                    matcher.bind(e.getKey(), e.getValue());
142                } catch (Exception ex) {
143                    LOG.error("Failed to bind " + e.getKey() + " to " + e.getValue(), ex);
144                }
145            }
146        }
147    }
148
149    @Nonnull
150    public Persister newPersister() {
151        Registry registry = new SuperClassLookupRegistry();
152        RegistryMatcher matcher = new SuperClassLookupRegistryMatcher();
153
154        Strategy strategy = new CycleStrategy(ID, REF);
155        strategy = new AnnotationStrategy(strategy);
156        strategy = new RegistryStrategy(registry, strategy);
157        Persister persister = new Persister(strategy, NoopFilter.INSTANCE, matcher);
158
159        addTransformers(matcher, persister, transformFactories);
160        addConverters(registry, persister, converterFactories);
161        return persister;
162    }
163
164    @Nonnull
165    public static Persister newDefaultPersister() {
166        PersisterFactory factory = new PersisterFactory();
167        factory.addConverters(ServiceLoader.load(ConverterFactory.class));
168        factory.addTransforms(ServiceLoader.load(TransformFactory.class));
169        return factory.newPersister();
170    }
171}
